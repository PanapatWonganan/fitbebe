        try {

                }
            });

                        xhr.setRequestHeader('Authorization', `Bearer ${token}`);
                    }
                });
                
                hls.loadSource(url);
                hls.attachMedia(videoRef.current!);
                
                hls.on(Hls.Events.MANIFEST_PARSED, () => {
                    setIsAuthorized(true);
                });
